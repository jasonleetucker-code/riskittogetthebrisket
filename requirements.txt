# ── Runtime dependencies ──────────────────────────────────────────────
#
# Single source of truth for the Python runtime.  Every environment
# (local dev, CI, production) installs from THIS file via
# ``pip install -r requirements.txt``.  Test/dev-only dependencies
# live in ``requirements-dev.txt`` which chains in this file via
# ``-r requirements.txt`` so there is no duplication.
#
# Versions use compatible-release (``~=``) pins so patch/minor
# bumps flow in but a surprise major-version bump cannot silently
# break CI or production.  To lock exact versions for reproducibility,
# run ``pip freeze > requirements.lock.txt`` after a clean install.
#
# Do NOT add test/dev-only dependencies here.  Add them to
# ``requirements-dev.txt`` instead.

# Web framework + ASGI server.  Pin syntax ``~=X.Y.0`` is
# compatible-release: patches flow (0.135.0 → 0.135.9) but a minor
# or major bump requires an explicit edit.  Minors bumped from the
# original PR pins to match what production + dev are actually
# running on 2026-04-24 — the narrower pins were authored three
# days earlier and fell behind the live env.
fastapi~=0.135.0
uvicorn~=0.44.0

# HTTP client used by scrapers, Sleeper client, news providers, etc.
requests~=2.31.0

# Browser automation (FootballGuys + DraftSharks scrapers).
playwright~=1.58.0

# Excel workbook parser (CSVs/Draft Data.xlsx → draft_data.csv).
openpyxl~=3.1.0

# Chat endpoint (``/api/chat``) — optional at runtime (server.py
# guards ``import anthropic`` with ImportError fallback) but declared
# here so production + CI install a consistent version and the feature
# is never silently disabled.
anthropic~=0.40.0

# DLF scraping: curl_cffi impersonates Chrome's TLS fingerprint so
# we can pass DLF's Cloudflare JS challenge (vanilla requests and
# Playwright both get blocked at the challenge page).
curl_cffi~=0.15.0

# beautifulsoup4 parses DLF's WPDataTable HTML + FantasyPros Fitzmaurice
# markup.
beautifulsoup4~=4.14.0

# Web Push (PWA notifications).  ``pywebpush`` handles the VAPID
# signing + RFC8291 payload encryption; ``cryptography`` is already
# pulled in transitively.  Used by ``src/api/push_delivery.py`` to
# fan out the daily signal-alerts digest + custom alert hits to
# subscribed devices.
pywebpush~=2.0

# nfl_data_py: deliberately NOT in this manifest.
#
# nfl_data_py 0.3.x pins pandas<2.0, which forces pandas-1.5.3 to
# build from source on Python 3.12 and fails (pkg_resources missing
# from modern setuptools).  Newer nfl_data_py releases that support
# pandas 2.x are not yet on PyPI as compatible-release pins our deploy
# manifest can use safely.
#
# To activate the nfl_data pipeline (after which value confidence
# intervals, usage signals, opportunity stats start producing real
# data instead of [] no-ops), install on the production VPS:
#
#     ssh dynasty@<vps>
#     source ~/.venvs/trade-calculator/bin/activate
#     pip install 'nfl_data_py>=0.4.0' --upgrade-strategy only-if-needed
#     systemctl restart dynasty
#
# Without nfl_data_py installed, every src/nfl_data/* fetch returns
# [] (guarded import in ingest._nfl_data_py_or_none()) — no crash,
# observable via /api/status.nflDataProvider.installed=False.
